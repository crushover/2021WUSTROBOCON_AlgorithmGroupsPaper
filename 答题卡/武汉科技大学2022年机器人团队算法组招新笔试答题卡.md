#include <stdio.h>
#include <stdlib.h>
#include"string.h"
int main(){
    int ans = 0;
    int bit = 0;
    int numsSize;
    int nums[100] = {2，2，2，4};
    numsSize = sizeof(nums)/sizeof(nums[0]);
    for(int i = 31; i >= 0; i--)
    {
        for(int j = 0; j < numsSize; j++)
        {
            bit += nums[j] >> i & 1;
        }
        ans = ans * 2 + bit % 3;
        bit = 0;
    }
